#include<iostream>
#include<windows.h>
#include<stdlib.h>
#include<stdio.h>
#include<time.h>
#include<conio.h>
#include<cmath>
#include <Windows.h>//windows编程头文件
using namespace std;

struct node {           //坐标定义
    int x;
    int y;
}snake[1000];

HANDLE hout= GetStdHandle(STD_OUTPUT_HANDLE);
COORD pos;
int h, l;                        //框架大小
int snake_length;                //蛇长度
node food;                        //食物坐标
int key=72,snake_speed;

void print_kuang(int h, int l)    //打印框架
{
    int i = 0, j = 0;
    for (i = 0; i <= h; i++)
        for (j = 0; j <= l; j++)
        {
            if (i == 0 || i == h || j == 0 || j == l)
            {
                if (j == l)
                    cout << '*' << endl;
                else
                    cout << '*';
            }
            else
                cout << ' ';
        }
}

void hide()                        //隐藏光标函数
{
    CONSOLE_CURSOR_INFO cursor_info = { 1,0 };
    SetConsoleCursorInfo(hout, &cursor_info);
}

void gotoxy(int x, int y)      //移动光标函数
{

    pos.X = x - 1;
    pos.Y = y - 1;    
    SetConsoleCursorPosition(hout, pos);
}

void print_food()        //打印食物函数
{
    int a, b, c ;
    while (1)
    {
        srand((unsigned)time(NULL));
        int a=rand()%(h+1)+2, b = rand() % 65 +1 ;
        food.x = b;
        food.y = a;
        c = 1;
        for (int k = 0; k <= snake_length - 1; k++)
        {
            if (snake[k].x == food.x&&snake[k].y == food.y)
            {
                c = 0;
                break;
            }
        }
        if (c!=0)
            break;
    }
    gotoxy(food.x, food.y);
    cout << '%';
}

void print_snake()                //首次打印蛇
{
    gotoxy(snake[0].x, snake[0].y);
    cout << "@";
    for (int i = 1; i <= snake_length - 1; i++)
    {
        gotoxy(snake[i].x, snake[i].y);
        cout << '*';
    }
}

bool is_correct()                //是否相撞
{
    if (snake[0].x == 1 || snake[0].y == 1 || snake[0].x == l || snake[0].y == h)
        return false;
    for (int i = 1; i < snake_length - 1; i++)
    {
        if (snake[0].x == snake[i].x&&snake[0].y == snake[i].y)
            return false;
    }
    return true;
}

void keyDown()
{
    int pre_key = key;
    if (_kbhit())
    {
        fflush(stdin);
        key = _getch();
        key = _getch();
    }
    bool e = false;
    node temp=snake[0],temp1=snake[snake_length-1];
    if (!(snake[0].x == food.x&&snake[0].y == food.y))     //没吃到食物
    {
        gotoxy(snake[snake_length-1].x, snake[snake_length-1].y);
        cout << ' ';
    }
    else
    {
        print_food();
        snake_length += 1;
        snake_speed -= 5;
    }
    for (int i = snake_length-1; i > 0; i--)
    {
        snake[i].x = snake[i - 1].x;
        snake[i].y = snake[i - 1].y;
    }
    snake[0] = temp;
    if (pre_key == 72 && key == 80)
        key = 72;
    if (pre_key == 80 && key == 72)
        key = 80;
    if (pre_key == 75 && key == 77)
        key = 75;
    if (pre_key == 77 && key == 75)
        key = 77;
    switch (key)
    {
    case 75:
        snake[0].x -= 1;//往左
        break;
    case 77:
        snake[0].x += 1;//往右
        break;
    case 72:
        snake[0].y--;//往上
        break;
    case 80:
        snake[0].y++;//往下
        break;
    }
    gotoxy(snake[0].x, snake[0].y);
    cout << '@';
    gotoxy(snake[1].x, snake[1].y);
    cout << '*';
}
